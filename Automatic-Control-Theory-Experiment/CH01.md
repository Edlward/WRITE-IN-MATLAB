>> G1 = tf(1, [1, 2, 1]);
>> G2 = tf(1, [1, 1]);
>> G3 = tf(1, [2, 1]);
>> G4 = tf(1, [1, 0]);
>> G12 = G1 + G2;
>> G34 = G3 - G4;
>> G = feedback(G12, G34, -1)

       2 s^4 + 7 s^3 + 7 s^2 + 2 s
